Use a modulo operator (position.z = velocity % segmentSize) to reset the position seamlessly inside segment boundaries, creating an endless movement illusion without asset accumulation.Responsiveness: Re-update camera aspect ratios and renderer sizes dynamically on window resize events.Reference Implementation Snippet (Vanilla JS)javascriptconst scene = new THREE.Scene();
scene.fog = new THREE.FogExp2(0x000000, 0.14);
const camera = new THREE.PerspectiveCamera(65, window.innerWidth / window.innerHeight, 0.01, 1000);
camera.position.set(0, 3.5, 6); camera.rotation.x = -0.42;

const renderer = new THREE.WebGLRenderer({ canvas: document.getElementById('grid-canvas'), antialias: true });
renderer.setSize(window.innerWidth, window.innerHeight);

const size = 40, segments = 40;
const grid = new THREE.Mesh(new THREE.PlaneGeometry(size, size, segments, segments), new THREE.MeshBasicMaterial({ color: 0x1f1f1f, wireframe: true }));
grid.rotation.x = -Math.PI / 2; scene.add(grid);

const clock = new THREE.Clock(), segSize = size / segments;
function tick() {
  requestAnimationFrame(tick);
  grid.position.z = (clock.getElapsedTime() * 1.8) % segSize;
  renderer.render(scene, camera);
}
tick();
